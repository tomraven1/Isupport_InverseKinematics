54% Map Minimum and Maximum Input Processing Function
55function y = mapminmax_apply(x,settings)
56  y = bsxfun(@minus,x,settings.xoffset);
57  y = bsxfun(@times,y,settings.gain);
58  y = bsxfun(@plus,y,settings.ymin);
59end
